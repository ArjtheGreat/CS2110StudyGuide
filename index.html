17. Write Defensive Code to Assert Preconditions of a Method


            // Method: withdraw(int amount)
            // Preconditions: amount > 0, amount <= balance
            
            public void withdraw(int amount) {
                if (amount <= 0) {
                    throw new IllegalArgumentException("Amount must be positive.");
                }
                if (amount > balance) {
                    throw new IllegalArgumentException("Insufficient funds.");
                }
                balance -= amount;
            }

22. Write a Class that Implements an Interface


            // Interface: ComplexNumber
            public interface ComplexNumber {
                double getReal();
                double getImaginary();
                ComplexNumber add(ComplexNumber other);
            }
    
            // Class: Complex
            public class Complex implements ComplexNumber {
                private double real;
                private double imaginary;
    
                public Complex(double real, double imaginary) {
                    this.real = real;
                    this.imaginary = imaginary;
                }
    
                public double getReal() {
                    return real;
                }
    
                public double getImaginary() {
                    return imaginary;
                }
    
                public ComplexNumber add(ComplexNumber other) {
                    return new Complex(this.real + other.getReal(), this.imaginary + other.getImaginary());
                }
            }

28. Classify a Cast


            // Example:
            Object obj = "Hello";
            
            // (i) Allowed at compile-time but could fail at runtime:
            Integer num = (Integer)obj;  // fails at runtime
    
            // (ii) Allowed at compile-time and will always succeed:
            String str = (String)obj;  // unnecessary
    
            // (iii) Allowed at compile-time but will always fail:
            Integer num2 = (Integer)"abc";  // fails at runtime
    
            // (iv) Not allowed at compile-time:
            // int num3 = (int)"abc";  // compile-time error

29. Use Bottom-Up Rule for Dynamic Dispatch


            // Example: Parent/Child class with dynamic dispatch
    
            public class Parent {
                public void speak() {
                    System.out.println("Parent speaks");
                }
            }
    
            public class Child extends Parent {
                public void speak() {
                    System.out.println("Child speaks");
                }
            }
    
            Parent p = new Child();
            p.speak();  // prints "Child speaks" due to dynamic dispatch

31. Constructor Delegates to Superclass


            // Example: 3DPoint constructor delegates to 2DPoint
    
            public class Point2D {
                protected int x, y;
    
                public Point2D(int x, int y) {
                    this.x = x;
                    this.y = y;
                }
            }
    
            public class Point3D extends Point2D {
                private int z;
    
                public Point3D(int x, int y, int z) {
                    super(x, y);  // delegate x and y to superclass
                    this.z = z;   // initialize z here
                }
            }

32. Method Delegates to Superclass


            // Example: negating 2D and 3D points
    
            public class Point2D {
                protected int x, y;
    
                public void negate() {
                    x = -x;
                    y = -y;
                }
            }
    
            public class Point3D extends Point2D {
                private int z;
    
                @Override
                public void negate() {
                    super.negate();  // delegate x and y negation to superclass
                    z = -z;          // negate z here
                }
            }

37. Evaluate Equals() Properties


            // Properties:
            // Reflexive: obj.equals(obj) -> true
            // Symmetric: obj1.equals(obj2) == obj2.equals(obj1)
            // Transitive: obj1.equals(obj2) && obj2.equals(obj3) -> obj1.equals(obj3)
    
            public class MyClass {
                @Override
                public boolean equals(Object obj) {
                    return this == obj;
                }
            }
    
            // Reflexive: true
            // Symmetric: true
            // Transitive: true
